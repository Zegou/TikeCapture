    function  changestatus(obj){
    var id = $(obj).attr('id');
        $.ajax({
        url:"{{url('account_service/accountservicechangestatus/')}}/"+id,
        type: "GET",
            dataType: 'json',
                success: function(data) {
                if(data.messages ==="ok"){
                notification = alertify.notify('Merci d\'avoir consulter le document ', 'success', 7, function () {
                location.reload();
                });
            }

        },
                error: function (jqXHR, textStatus, errorThrown) {
                alert("Probl&egrave;me lors de l'envoi veuillez Contacter l'adminstrateur");

                }
        })
    }

        $.ajax({

            dataType: 'html',

                $("div#detailsaccountserviceform").html(data);

            },
            error: function (jqXHR, textStatus, errorThrown) {
                alert("Probl&egrave;me lors de l'envoi veuillez Contacter l'adminstrateur");

            }
        })
    }

    // Envoyer les documents au service Achat
    $("form#sendbackpurchaseserviceform_id").on("submit", function (event) {
        $.ajaxSetup({
            headers: {
                'X-CSRF-TOKEN': $('meta[name="csrf-token"]').attr('content')
            }
        });
        event.preventDefault();
        $.ajax({
            url:"{{url('accountingservice/accountingservicesendback')}}",
            method:"POST",
            data:new FormData(this),
            contentType: false,
            cache: false,
            processData: false,

            success: function (data) {
                if (data.messages === "ok") {

                    notification = alertify.notify('Votre document A été bien renvoyé au service Achat', 'success', 7, function () {
                        setTimeout(function(){   window.location.reload(); }, 4000);


                    });
                } else if (data.messages === "erreur") {

                    notification = alertify.notify("Ce dossier n'existe  pas au service achat", "error", 7, function () {

                    });
                }
